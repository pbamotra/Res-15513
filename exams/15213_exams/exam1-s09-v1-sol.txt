void mystery_sort (long* array, long len)
{
  long a, b, tmp;
  
  while(len>0)
  {
  
    a = 0;

    for(b=len-1; b>0; b--)
    {
      if(array[b]>array[a])
      {
        a=b;
      }
    }

    len--;
    tmp = array[len];
    array[len] = array[a];
    array[a] = tmp;
}
